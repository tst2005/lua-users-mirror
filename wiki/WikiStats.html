The stats are manually collected as follows: 


wget -r -np -D lua-users.org --html-extension http://lua-users.org/wiki/

cd lua-users.org/wiki

rm index.html

echo "pages: `ls | wc -l`"

find -name '*.html' -exec lynx -dump -nolist '{}' > text_dump ';'

echo "bytes: `wc -c text_dump`"
