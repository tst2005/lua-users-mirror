function build_message_box(gui_builder) local my_dialog = gui_builder:dialog() my_dialog:set_title("Message Box")

local my_label = gui_builder:label() my_label:set_font_size(20) my_label:set_text("Hello, world!") my_dialog:add(my_label)

local my_button = gui_builder:button() my_button:set_title("OK") my_dialog:add(my_button)

return my_dialog end

build_message_box = gui:dialog "Message Box" { gui:label "Hello, world!" { font_size = 20 }; gui:button "OK" { }; }
